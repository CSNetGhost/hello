# --------------------------------------------------------------------------------------------------
# This file inlcudes various "simple" signatures (radar, ir, and optical)
#
#   SIMPLE_FIGHTER_RCS which is a 5 square meter target, typical for a fighter-sized aircraft
#   SIMPLE_FIGHTER_IR_SIG which is a 200 watts/steradian infrared target, typical for non-afterburning aircraft
#   SIMPLE_FIGHTER_IR_SIG_AFTERBURNER which is a 1000 watts/steradian infrared target, typical for afterburning aircraft
#   SIMPLE_FIGHTER_OPTICAL_SIG which is a 3 square meter optical target, typical for a fighter-sized aircraft
#
#   SIMPLE_1M_RCS which is a 1 square meter target
#   SIMPLE_100WSR_IR_SIG which is a 100 watts/steradian infrared target
#   SIMPLE_1M_OPTICAL_SIG which is a 1 square meter optical target
#
# --------------------------------------------------------------------------------------------------

# Radar cross sections

radar_signature SIMPLE_FIGHTER_RCS
   constant 5 m^2
end_radar_signature

radar_signature SIMPLE_1M_RCS
   constant 1 m^2
end_radar_signature

# --------------------------------------------------------------------------------------------------

# Infrared signatures

infrared_signature SIMPLE_FIGHTER_IR_SIG
   state default
      band default
         constant 200 w/sr
end_infrared_signature

infrared_signature SIMPLE_FIGHTER_IR_SIG_AFTERBURNER
   state default
      band default
         constant 1000 w/sr
end_infrared_signature

infrared_signature SIMPLE_100WSR_IR_SIG
   state default
      band default
         constant 100 w/sr
end_infrared_signature

# --------------------------------------------------------------------------------------------------

# Optical signatures

optical_signature SIMPLE_FIGHTER_OPTICAL_SIG
   constant 3 m^2
end_optical_signature

optical_signature SIMPLE_1M_OPTICAL_SIG
   constant 1 m^2
end_optical_signature

# --------------------------------------------------------------------------------------------------
