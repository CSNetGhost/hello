script void MOE_DataTestAction(string aLatitude,
                               string aLongitude,
                               string aAltitude,
                               string aID,
                               string aValue)
   ExpectBool((double)aAltitude > 0.0, true, "Should see terrain altitude");
end_script

script void RunTests()
   Path path = Path.GetWorkingDirectory();
   path.Join("terraintest");
   
   path.Join("test_coverage_sc_data.csv");
   CheckDataFile(path, "MOE data", "MOE_DataTestAction", 121);
   path.Up();
   
   FinalAssessment();
   
   RemoveCoverageOutputDir(path);
end_script

observer
   enable SIMULATION_COMPLETE RunTests
end_observer
