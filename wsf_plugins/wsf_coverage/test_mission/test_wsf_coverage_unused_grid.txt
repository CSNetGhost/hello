grid test_grid WSF_LAT_LON_GRID
   altitude 1 m msl
   latitude_span 1s 1n
   longitude_span 1w 1e
   latitude_spacing 0.2 deg
   longitude_spacing 0.2 deg
   asset TEST_GRID_PLATFORM none
   grid_data_file unusedgrid/data.csv
end_grid

script_variables
end_script_variables

script void RunTests()
   int count = WsfSimulation.PlatformCount();
   for (int i = 0; i < count; i += 1)
   {
      if(WsfSimulation.PlatformEntry(i).Type() == "TEST_GRID_PLATFORM")
      {
         gAllPassed = false;
         writeln("-FAIL- A grid platform was created for an unused grid.");
         break;
      }
   }
   
   FinalAssessment();
end_script

observer
   enable SIMULATION_COMPLETE RunTests
end_observer
