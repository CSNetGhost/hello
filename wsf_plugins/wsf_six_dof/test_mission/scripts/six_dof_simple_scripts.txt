script void ActivateSequencer(WsfPlatform aPlatform, string aSequencerName)
   WsfSixDOF_Mover mover = (WsfSixDOF_Mover)aPlatform.Mover();
   if(mover.IsValid())
   {
      mover.ActivateSequencer(aSequencerName);
   }
   else
   {
      writeln("--ERROR-- Not a P6DOF Mover.");
   }
end_script

script void AddFuel_lbs(WsfPlatform aPlatform, double aFuelMass_lbs)
   WsfSixDOF_Mover mover = (WsfSixDOF_Mover)aPlatform.Mover();
   if(mover.IsValid())
   {
      mover.AddFuel(aFuelMass_lbs * Math.KG_PER_LB());
   }
   else
   {
      writeln("--ERROR-- Not a P6DOF Mover.");
   }
end_script

script double GetFuelInTank_lbs(WsfPlatform aPlatform, string aTankName)
   WsfSixDOF_Mover mover = (WsfSixDOF_Mover)aPlatform.Mover();
   if(mover.IsValid())
   {
      return mover.GetFuelInTank(aTankName) * Math.LB_PER_KG();
   }
   else
   {
      writeln("--ERROR-- Not a P6DOF Mover.");
      return 0.0;
   }
end_script
