script void RollToBankAngle(WsfPlatform aPlatform, double bankAngle_deg)
   WsfSixDOF_Mover mover = (WsfSixDOF_Mover)aPlatform.Mover();
   if(!mover.IsNull())
   {
      mover.SetAutopilotRollAngle(bankAngle_deg);
   }
   else
   {
      writeln("--ERROR-- Not a P6DOF Mover.");
   }
end_script
