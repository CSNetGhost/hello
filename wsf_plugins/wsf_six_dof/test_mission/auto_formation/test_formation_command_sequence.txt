execute at_time 1 min absolute
   // Create the replacement unit formation.
   WsfSixDOF_Formation newUnit = WsfSixDOF_FormationManager.CreateFormation("unit", "replacement");
   newUnit.SetMember("india");

   // check the initial state
   WsfSixDOF_Formation yankeeBeta = WsfSixDOF_FormationManager.GetFormation("yankee.beta");
   allPassed = ExpectBool(yankeeBeta.IsValid(), true, "yankee.beta should exist") && allPassed;
   allPassed = ExpectBool(yankeeBeta.IsAttached(), true, "yankee.beta should be attached") && allPassed;
   allPassed = ExpectBool(yankeeBeta.GetSubFormation("two").IsValid(),
              true,
              "yankee.beta should have two as a subformation") && allPassed;
   allPassed = ExpectBool(yankeeBeta.GetSubFormation("replacement").IsValid(),
              false,
              "yankee.beta should not have replacement as a subformation") && allPassed;
   
   WsfSixDOF_Formation yankeeAlternate = WsfSixDOF_FormationManager.GetFormation("yankee.alternate");
   allPassed = ExpectBool(yankeeAlternate.IsValid(), false, "yankee.alternate should not exist") && allPassed;
   
   WsfSixDOF_Formation yankeeBetaTwo = WsfSixDOF_FormationManager.GetFormation("yankee.beta.two");
   allPassed = ExpectBool(yankeeBetaTwo.IsValid(), true, "yankee.beta.two should exist") && allPassed;
   
   WsfSixDOF_Formation replacement = WsfSixDOF_FormationManager.GetFormation("replacement");
   allPassed = ExpectBool(replacement.IsValid(), true, "replacement should exist") && allPassed;
   
   WsfSixDOF_Formation unitTwo = WsfSixDOF_FormationManager.GetFormation("two");
   allPassed = ExpectBool(unitTwo.IsValid(), false, "two should not exist at top level") && allPassed;
end_execute

execute at_time 2 min absolute
   WsfSixDOF_FormationCommandSequence sqn = WsfSixDOF_FormationCommandSequence.Construct();
   
   WsfSixDOF_FormationDetachCommand detach = WsfSixDOF_FormationDetachCommand.Construct();
   sqn.AppendCommand(detach);
   
   WsfSixDOF_FormationRenameCommand rename = WsfSixDOF_FormationRenameCommand.Construct("alternate");
   rename.SetConstraint(WsfSixDOF_FormationCommandConstraint.AT_RELATIVE_TIME(60.0));
   sqn.AppendCommand(rename);
   
   WsfSixDOF_FormationRemoveSubCommand remove = WsfSixDOF_FormationRemoveSubCommand.Construct("two");
   remove.SetConstraint(WsfSixDOF_FormationCommandConstraint.AT_RELATIVE_TIME(60.0));
   sqn.AppendCommand(remove);
   
   WsfSixDOF_FormationAddSubCommand add 
      = WsfSixDOF_FormationAddSubCommand.Construct("replacement",
                                            WsfSixDOF_FormationManager.GetFormation("yankee.beta.two").GetOffset());
   add.SetConstraint(WsfSixDOF_FormationCommandConstraint.AT_RELATIVE_TIME(60.0));
   sqn.AppendCommand(add);
   
   sqn.AppendCommand(WsfSixDOF_FormationAttachCommand.Construct());
   
   WsfSixDOF_Formation yankeeBeta = WsfSixDOF_FormationManager.GetFormation("yankee.beta");
   yankeeBeta.ExecuteCommand(sqn, WsfSixDOF_FormationCommandConstraint.AT_RELATIVE_TIME(60.0)); 
end_execute
