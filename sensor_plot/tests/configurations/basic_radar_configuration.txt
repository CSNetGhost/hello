# A very simple radar configuration
# with range of 50 km
sensor RADAR_TYPE WSF_RADAR_SENSOR 
   one_m2_detect_range 50 km
   frame_time 1 s
   reports_location
   azimuth_field_of_view -100.0 deg 100.0 deg
   elevation_field_of_view 5 deg 70 deg
   transmitter 
      frequency 1 ghz
      power 1 milliwatts
   end_transmitter
end_sensor

platform_type SENSOR_PLATFORM_TYPE WSF_PLATFORM
   sensor radar-1 RADAR_TYPE
      on
   end_sensor
end_platform_type

platform sensor-1 SENSOR_PLATFORM_TYPE 
end_platform
