#                      Define the parameters for the plot

vertical_coverage
  # output_base defines the base part of the output files.
  output_base    multi-beam

  # elevation_range defines the range of elevations that will be processed.
  # The default is -5 deg to 89.9 deg
  elevation_range -2 deg 89.9 deg

  # elevation_step defines the increment between elevations.
  # The default is 0.1 deg
  elevation_step 0.05 deg

  # maximum_range defines the maximum range that will be checked.
  # The default is to use the value from the sensors 'maximum_range' value.
  # If this value is not supplied and if the sensor does not have a value
  # for 'maximum_range' then 2000 km will be used.
  maximum_range 450 km

  # altitude_units defines the altitude units to be used the resulting plot.
  # The default is 'm'
  altitude_units ft

  # range_units defines the range units to be used in the resulting plot.
  # The default is 'm'
  range_units    nm
end_vertical_coverage
