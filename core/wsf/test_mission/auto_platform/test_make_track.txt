// MakeTrack should create a track that Is3D.
//
// LocationValid exposes the Is3D track parameter to script.
//

platform test_plat WSF_PLATFORM
   execute at_time 1 s absolute
      WsfTrack track = PLATFORM.MakeTrack();
      if (track.LocationValid())
      {
         writeln("-PASS-");
      }
      else
      {
         writeln("-FAIL-");
      }
   end_execute
end_platform
