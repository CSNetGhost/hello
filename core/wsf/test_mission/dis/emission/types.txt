antenna_pattern RADAR_ANTENNA
   uniform_pattern
      azimuth_beamwidth   4 deg
      elevation_beamwidth 2 deg
      peak_gain           35 db
end_antenna_pattern

sensor RADAR WSF_RADAR_SENSOR
   frame_time 1 sec

   beam 1
      scan_mode          azimuth

      transmitter
         antenna_pattern RADAR_ANTENNA
         antenna_tilt    2.0 deg
         power           100 kw
         frequency       3 ghz
      end_transmitter
      one_m2_detect_range 200 nm

      swerling_case           1
   end_beam

   hits_to_establish_track 3 5
   hits_to_maintain_track  1 1
   compute_measurement_errors true

   reports_range
   reports_bearing
   reports_elevation
end_sensor

platform_type RADAR WSF_PLATFORM
   icon Ground_Radar

   sensor radar RADAR
      ignore_same_side
      processor data_mgr
   end_sensor

   processor data_mgr WSF_TRACK_PROCESSOR
   end_processor

end_platform_type

// ----------------------------------------------------------------------------

antenna_pattern PASSIVE_ANTENNA
   uniform_pattern
      peak_gain 10 db
end_antenna_pattern

sensor AIRCRAFT_SENSOR WSF_PASSIVE_SENSOR
   frame_time 1 sec
   receiver
      noise_power      -150 dbw
   end_receiver
   detection_threshold 12.8 db

   frequency_band 1 ghz 18 ghz

   reports_bearing
   reports_range
end_sensor

radar_signature AIRCRAFT_RADAR_SIG
   constant 1 m^2
end_radar_signature

infrared_signature AIRCRAFT_INFRARED_SIG
   constant 1000 w/sr
end_infrared_signature

optical_signature AIRCRAFT_OPTICAL_SIG
   constant 10 m^2
end_optical_signature

platform_type AIRCRAFT WSF_PLATFORM
   icon  Wedge
   radar_signature    AIRCRAFT_RADAR_SIG
   infrared_signature AIRCRAFT_INFRARED_SIG
   optical_signature  AIRCRAFT_OPTICAL_SIG

   mover WSF_AIR_MOVER
   end_mover

   sensor passive AIRCRAFT_SENSOR
      on
      ignore_same_side
      processor data_mgr
   end_sensor

   processor data_mgr WSF_TRACK_PROCESSOR
   end_processor

end_platform_type
