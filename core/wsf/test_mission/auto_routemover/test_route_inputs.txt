# Test for graceful handling
# of an empty route in a route mover.
#
#
#       A Failure results in a "***** ERROR:" compile message.
#
#       Currently no method exist to trap the "ERROR" message to output the test "-FAIL-" message
#       therefore the absence of "-PASS-" is a test failure.

expect_input_error false

# Valid input
mover test_mover_empty_route WSF_GROUND_MOVER
   route
   end_route
end_mover

# Valid input
mover test_mover_w_route WSF_GROUND_MOVER
   route
      position 0n 0w
      position 1n 0w
   end_route
end_mover
