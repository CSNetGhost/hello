# Tests that a sensor may report nothing

script_variables
   bool pass = false;
end_script_variables

platform p WSF_PLATFORM
   add sensor geo WSF_GEOMETRIC_SENSOR
      on
      frame_time 1 sec
      reports_nothing
   end_sensor
end_platform

platform tgt WSF_PLATFORM
end_platform

script void SensorDetectionAttempt(WsfPlatform aPlatform, WsfSensor aSensor, WsfPlatform aTarget, WsfSensorInteraction aResult)
   pass = true;
end_script

script void SimulationComplete()
   if (!pass) { writeln("-FAIL-"); }
   else       { writeln("-PASS-"); }
end_script

observer
   enable SENSOR_DETECTION_ATTEMPT
   enable SIMULATION_COMPLETE
end_observer

end_time 1 sec
