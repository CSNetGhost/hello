# Tests the 'negative' command for zones
zone z1
   circular
   position 0n 0e
   maximum_radius 10 m
end_zone
zone z2
   circular
   position 0n 0e
   maximum_radius 10 m
   negative
end_zone

platform inside WSF_PLATFORM
   position 0.0n 0.0e
   execute at_time 1 s absolute
      WsfGeoPoint loc = Location();
      if (loc.WithinZone("z1"))
      {
         writeln("-PASS-");
      }
      else
      {
         writeln("-FAIL- Test 1");
      }
      if (! loc.WithinZone("z2"))
      {
         writeln("-PASS-");
      }
      else
      {
         writeln("-FAIL- Test 2");
      }
   end_execute
end_platform

platform outside WSF_PLATFORM
   position 1n 1e
   execute at_time 1 s absolute
      WsfGeoPoint loc = Location();
      if (! loc.WithinZone("z1"))
      {
         writeln("-PASS-");
      }
      else
      {
         writeln("-FAIL- Test 3");
      }
      if (loc.WithinZone("z2"))
      {
         writeln("-PASS-");
      }
      else
      {
         writeln("-FAIL- Test 4");
      }
   end_execute
end_platform

end_time 10 s
