# should be a script compilation error due to unreachable code after a return.
script_expect_error true
script void Unreachable()
   writeln("reachable-code");
   return;
   writeln("unreachable-code-here");
end_script

# should be a script compilation error due to all code paths returning before the declaration of 'int i'.
script_expect_error true
script int UnreachableIfElse(bool b)
   if (b)
      return 1;
   else
      return 2;
   int i = 3; // <- error should point to this line
   // further unreachable code that should not generate additional unreachable-code errors.
   int q = i;
   return q;
end_script
