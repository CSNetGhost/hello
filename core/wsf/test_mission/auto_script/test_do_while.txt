# Test_do_while.txt
# This test will make sure the do-while block breaks at the correct time.

on_initialize
   int i   = 0;

   do
   {
      i = i + 1;
   } while (i < 10);

   if (i != 10)
   {
      writeln("-FAIL- : The loop block did not break at the correct time, per the while definition statement.");
   }
   else
   {
      writeln("-PASS-");
   }
end_on_initialize
