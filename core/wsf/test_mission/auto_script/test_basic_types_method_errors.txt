# Test for expected script compilation errors when invoking script Object methods on basic types

script_variables
   int    i;
   double d;
   bool   b;
   string s;
end_script_variables

script_expect_error true
script void IntToString()
   i.ToString();
end_script

script_expect_error true
script void IntIsValid()
   i.IsValid();
end_script

script_expect_error true
script void IntIsNull()
   i.IsNull();
end_script

script_expect_error true
script void DoubleToString()
   d.ToString();
end_script

script_expect_error true
script void DoubleIsValid()
   d.IsValid();
end_script

script_expect_error true
script void DoubleIsNull()
   d.IsNull();
end_script

script_expect_error true
script void BoolToString()
   b.ToString();
end_script

script_expect_error true
script void BoolIsValid()
   b.IsValid();
end_script

script_expect_error true
script void BoolIsNull()
   b.IsNull();
end_script

script_expect_error true
script void StringToString()
   s.ToString();
end_script

script_expect_error true
script void StringIsValid()
   s.IsValid();
end_script

script_expect_error true
script void StringIsNull()
   s.IsNull();
end_script
