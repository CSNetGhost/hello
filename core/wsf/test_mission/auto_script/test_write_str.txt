# test_write_str.txt
# Test the write_str method.

include test_functions.txt

on_initialize
   bool pass = true;

   string str1 = "Testing the write_str method 1 2 3 4 5";
   string str2 = write_str("Testing ", "the ", "write_str method ", 1, " ", 2, " ", 3, " ", 4, " ", 5);
   if (str1 != str2)
   {
      pass = false;
      writeln("-FAIL- : write_str method did not produce the correct string given the input.");
   }

   if (write_str("Testing ", "the ", "write_str method ", 1, " ", 2, " ", 3, " ", 4, " ", 5) != str2)
   {
      pass = false;
      writeln("-FAIL- : write_str method did not produce the correct string given the input.");
   }

   writePass(pass);
end_on_initialize
