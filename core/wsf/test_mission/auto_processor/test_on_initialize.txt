platform tester WSF_PLATFORM
	add processor script_proc WSF_SCRIPT_PROCESSOR
		update_interval 1 sec
		script_variables
			int x = 5;
			int expected_value = 3;
		end_script_variables
		on_initialize
			extern int x;
			x = 3;
		end_on_initialize
		on_update
			extern int x;
			extern int expected_value;
			if (x == expected_value) {
				writeln("Test Passed");
			}
			else{
				writeln("-FAIL- X was never assigned");
			}
			WsfSimulation.Terminate();
		end_on_update
	end_processor
end_platform
end_time 200 sec
