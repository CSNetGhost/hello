antenna_pattern EW_RADAR_ANTENNA
   uniform_pattern
      azimuth_beamwidth   4 deg
      elevation_beamwidth 2 deg
      peak_gain           35 db
end_antenna_pattern

sensor EW_RADAR WSF_RADAR_SENSOR
   frame_time 1 sec

   beam 1
      scan_mode          azimuth

      transmitter
         antenna_pattern EW_RADAR_ANTENNA
         antenna_tilt    2.0 deg
         power           100 kw
         frequency       3 ghz
      end_transmitter
      one_m2_detect_range 200 nm

      swerling_case           1
   end_beam

   hits_to_establish_track 3 5
   hits_to_maintain_track  1 1
   compute_measurement_errors true

   reports_range
   reports_bearing
   reports_elevation
end_sensor

platform_type EW_RADAR WSF_PLATFORM
   icon Ground_Radar
   
   sensor radar EW_RADAR
      ignore_same_side
      processor data_mgr
   end_sensor
   
   processor data_mgr WSF_TRACK_PROCESSOR
   end_processor

end_platform_type
