///////////////////////////Static Eval Triggers///////////////////////////
// These triggers are static definitions only used to add additional triggers
// to a composite trigger at runtime.
cyber_trigger static_global_trigger WSF_CYBER_TRIGGER
   debug
   execute_trigger false
   
   script bool OnEvaluate()
      static_global_evals = (static_global_evals + 1);
      if(DEBUG_STATIC_GLOBAL_EVAL)
      {
         writeln("static_global_trigger evaluating at t = " + (string)TIME_NOW);
      }
      if(TIME_NOW < 121.0)
      {
         return false;
      }
      return true;
   end_script 
end_cyber_trigger
