file(GLOB SRCS *.cpp *.hpp)
if(GTest_FOUND)
   add_executable(geodata_test ${SRCS})

   target_link_libraries(geodata_test PUBLIC
      geodata
      GTest::GTest
      GTest::Main
      ${SWDEV_THREAD_LIB}
      ${SWDEV_DL_LIB}
   )
   add_test(NAME "geodata" COMMAND geodata_test "${CMAKE_CURRENT_SOURCE_DIR}/data")
   set_property(TARGET geodata_test PROPERTY FOLDER UnitTests)
   # the VS_DEBUGGER_COMMAND_ARGUMENTS property makes the tests executable from VS and requires cmake 3.13, this will have no effect otherwise
   set_property(TARGET geodata_test PROPERTY VS_DEBUGGER_COMMAND_ARGUMENTS "${CMAKE_CURRENT_SOURCE_DIR}/data")
endif()
